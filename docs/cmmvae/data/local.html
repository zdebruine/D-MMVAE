 43    def setup(self, stage):
 44        """
 45        Sets up the data pipelines based on the current stage of the training process.

 47        Args:

 49            
 50        """
 51        if stage in (TrainerFn.FITTING,):
 52            self._train_datapipe = self.train_datapipe()
 53            self._val_datapipe = self.val_datapipe()
 54        elif stage in (TrainerFn.VALIDATING,):
 55            self._val_datapipe = self.val_datapipe()
 56        elif stage in (TrainerFn.PREDICTING, TrainerFn.TESTING):
 57            self._test_datapipe = self.test_datapipe()
