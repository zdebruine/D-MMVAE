275    def forward(self, x: torch.Tensor, metadata: pd.DataFrame, condition: Optional[str] = None):
276        """
277        Run forward pass through each metadata label specific condition layer. The metadata is 
278        grouped by the batch_key and the indices are used to mask the input tensor. The resulting
279        Tensors are summed to form the output batch maintaining original sample order.
280        
281        Args:
282            x (torch.Tensor): Input tensor.
283            metadata (pd.DataFrame): Metadata dataframe with corresponding batch_key
284            condition (str, optional): If available, all samples passed through this condition.
285        
286        Returns:
287            torch.Tensor: Output tensor of same shape as input.
288        """
289        if condition:
290            return self.conditions[condition](x), metadata
291        
292        active_conditions = set()
293        xhat = torch.zeros_like(x)
294        
295        for condition, group_df in metadata.groupby(self.batch_key):
296            condition = self.format_condition_key(condition)
297            mask = torch.zeros(len(metadata), dtype=int, device=x.device)
298            mask[group_df.index] = 1
299            xhat = xhat + self.conditions[condition](x * mask.unsqueeze(1))
300            active_conditions.add(condition)
301            
302        self.active_condition_modules = active_conditions
303        
304        return xhat

307class ConditionalLayers(nn.Module):
308    
309    """
310    Iterates over all ConditionLayer's allowing for shuffling or sequential iteration.
311    
312    Attributes:
313        layers (nn.ModuleList): nn.ModuleList of :class:`ConditionLayer`'s
314        selection_order (torch.Tensor): torch.Tensor representing selection order.
315    """
316    
317    def __init__(
318        self,
319        conditional_paths: dict[str, str],
320        fc_block_config: FCBlockConfig,
321        selection_order: Optional[list[str]] = None,
322    ):
323        """
324        Initialize :class:`ConditionLayer`'s for each batch_key.
325        
326        .. note::
327            All :class:`ConditionLayer`'s are instantiated from the same configuration. 
328            
329            Each key in conditional_paths is a :class:`ConditionLayer`.
330            
331        Args:
332            conditional_paths (dict[str, str]): Dictionary of batch_keys and paths to csv's with their respective unique conditions.
333            fc_block_config (cmmvae.modules.base.FCBlockConfig): Configuration for all ConditionLayer's.
334            selection_order (Optional[list[str]]: Optional list of batch_keys to order forward pass on respective ConditionLayer's.
335        """
336        super(ConditionalLayers, self).__init__()
337        
338        if not selection_order:
339            selection_order = list(conditional_paths.keys())
340            self.shuffle_selection_order = True
341        else:
342            self.shuffle_selection_order = False
343            
344        self.selection_order = torch.arange(0, len(selection_order), dtype=torch.int32, requires_grad=False)
345        
346        self.layers = nn.ModuleList([
347            ConditionalLayer(batch_key, conditional_paths[batch_key], fc_block_config)
348            for batch_key in selection_order
349        ])
350        
351    def forward(self, x: torch.Tensor, metadata: pd.DataFrame):
352        """
353        Run forward pass through each :class:`ConditionLayer` either shuffled or in-order.
354        
355        Args:
356            x (torch.Tensor): Input tensor
357            metadata (pd.DataFrame): Input metadata
358        
359        Returns:
360            torch.Tensor: Resulting torch.Tensor
361        """
362        order = self.selection_order
363        
364        if self.shuffle_selection_order:
365            permutation = torch.randperm(self.selection_order.size(0))
366            order = order[permutation]
367            
368        for idx in order:
369            x = self.layers[idx](x, metadata)
370        
371        return x

377class Encoder(nn.Module):
378    """
379    Encoder module for a Variational Autoencoder (cmmvae.modules.VAE) with flexible configurations.
380
381    Attributes:
382        encoder (cmmvae.modules.base.FCBlock): Fully connected block for encoding input features.
383        mean_encoder (nn.Linear): Linear layer to compute the mean of the latent variables.
384        var_encoder (nn.Linear): Linear layer to compute the variance of the latent variables.
385        z_transformation (Callable): Transformation applied to the latent variables. Defaults to softmax for log-normal distribution.
386        var_activation (Callable): Activation function for the variance. Defaults to torch.exp.
387        var_eps (float): Small epsilon value for numerical stability.
388        return_dist (bool): Whether to return the distribution object.
389    """
390    
391    def __init__(
392        self,
393        latent_dim: int,
394        fc_block_config: FCBlockConfig,
395        distribution: Union[Literal['ln'], Literal['normal']] = 'normal',
396        return_dist: bool = False,
397        var_eps: float = 1e-4, # numerical stability
398    ):
399        """
400        Initialize Encoder nn.Module from configuration and set defaults.
401        
402        Args:
403            latent_dim (int): Size of latent dimension.
404            fc_block_config (`FCBlockConfig`): Configuration for fc block.
405            distribution (Union[Literal['ln'], Literal['normal']], optional): Type of distribution for the latent variables. Defaults to 'normal'.
406            return_dist (bool, optional): Whether to return the distribution object. Defaults to False.
407            var_eps (float, optional): Small epsilon value for numerical stability in variance calculation. Defaults to 1e-4.
408        """
409        super().__init__()
410        self.fc = FCBlock(fc_block_config)
411
412        # Get hidden and latent dimenision from layer dim list
413        n_hidden = fc_block_config.layers[-1]
414        
415        # Linear layer to compute the mean of the latent variables
416        self.mean_encoder = nn.Linear(n_hidden, latent_dim)
417        
418        # Linear layer to compute the variance of the latent variables
419        self.var_encoder = nn.Linear(n_hidden, latent_dim)
420        
421        # Transformation for latent variables (softmax for log-normal distribution, identity otherwise)
422        self.z_transformation = nn.Softmax(dim=-1) if distribution == "ln" else _identity
423        # Small epsilon value for numerical stability
424        self.var_eps = var_eps
425        
426        # Whether to return the distribution object
427        self.return_dist = return_dist
428    
429    @property
430    def n_layers(self) -> int:
431        """Number of layer's."""
432        return self.fc.config.n_layers
433    
434    def encode(self, x: torch.Tensor):
435        """
436        Encode input and return output and list of hidden representations
437        
438        Args:
439            x (torch.Tensor): Input tensor
440        
441        Returns:
442            torch.Tensor: Output tensor
443        """
444        return self.fc(x)
445    
446    def forward(self, x: torch.Tensor):
447        """
448        Forward pass through the encoder.
449
450        Args:
451            x (torch.Tensor): Input tensor.
452
453        Returns:
454            Union[Tuple[torch.distributions.Normal, torch.Tensor], Tuple[torch.Tensor, torch.Tensor, torch.Tensor]]: 
455                If return_dist is True, returns the distribution and latent variables.
456                Otherwise, returns the mean, variance, and latent variables.
457        """
458        # Encode the input features
459        encoded = self.encode(x)
460        if isinstance(encoded, tuple):
461            q, hidden_representations = encoded
462        else:
463            hidden_representations = []
464            q = encoded
465        
466        # Compute the mean of the latent variables
467        q_m = self.mean_encoder(q)
468        # Compute the variance of the latent variables and add epsilon for numerical stability
469        q_v = torch.exp(self.var_encoder(q)) + self.var_eps
470        
471        # Create a normal distribution with the computed mean and variance
472        dist = Normal(q_m, q_v.sqrt())
473        
474        # Sample the latent variables and apply the transformation
475        latent = self.z_transformation(dist.rsample())
476        
477        if self.return_dist:
478            return dist, latent, hidden_representations
479        
480        return q_m, q_v, latent, hidden_representations

186    def _make_layer(self, n_in: int, n_out: int, use_batch_norm: bool, use_layer_norm: bool, activation_fn: Optional[Type[nn.Module]], dropout_rate: float) -> nn.Sequential:
187        """
188        Create a single fully connected layer.
189
190        Args:
191            n_in (int): Number of input units.
192            n_out (int): Number of output units.
193            use_batch_norm (bool): Whether to use batch normalization.
194            use_layer_norm (bool): Whether to use layer normalization.
195            activation_fn (Optional[Type[nn.Module]]): Activation function.
196            dropout_rate (float): Dropout rate.
197
198        Returns:
199            nn.Sequential: A sequence of layers comprising the fully connected layer.
200        """
201        layers = [nn.Linear(n_in, n_out)]
202
203        if use_batch_norm:
204            layers.append(nn.BatchNorm1d(n_out, momentum=0.01, eps=0.001))
205        if use_layer_norm:
206            layers.append(nn.LayerNorm(n_out, elementwise_affine=False))
207        if activation_fn is not None:
208            layers.append(activation_fn(dim=1) if issubclass(activation_fn, nn.Softmax) else activation_fn())
209        if dropout_rate > 0:
210            layers.append(nn.Dropout(p=dropout_rate))
211
212        return nn.Sequential(*layers)

215        """
216        Forward pass through the fully connected block.
217
218        Args:
219            x (torch.Tensor): Input tensor.
220
221        Returns:
222            Union[torch.Tensor, tuple[torch.Tensor, list[torch.Tensor]]]: Output tensor or tuple of output and hidden representations if required.
223        """
224        if self.can_bypass:
225            return self.fc_layers(x)

229            for name, sublayer in layer.named_children():
230                x = sublayer(x)
231                if name == '2' and self.config.return_hidden[i]:
232                    hidden_representations.append(x)
233        return x, hidden_representations

 65        return_hidden: Union[bool, List[bool]] = False

 68        
 69        .. note::
 70            The FCBlock zips the layers into n_in, n_out pairings ([10, 20, 30] -> [(10, 20), (20, 30)]). If 
 71            a single layer size is provided, it's size is duplicated to form one pairing ([10] -> [(10, 10)]).
 72

 74            or a single value that is duplicated for all layers.
 75

 80            use_layer_norm (Union[bool, List[bool]], optional): Whether to use layer normalization for each layer. Defaults to False.
 81            activation_fn (Union[Optional[Type[nn.Module]], List[Optional[Type[nn.Module]]]], optional): Activation function(s) for each layer. Defaults to None.
 82            return_hidden (Union[bool, List[bool]], optional): Whether to aggregate and return hidden representations. Defaults to False.
 83        """
 84        super().__init__()
 85        
 86        # Assert layers is a list
 87        assert isinstance(layers, list), f"layers must be a list found type: {type(layers)}"
 88        # Assert all values are integer objects greater than 0
 89        assert all(isinstance(layer, int) and layer > 0 for layer in layers), "layers must be positive integers"
 90        if len(layers) == 1:
 91            layers = layers * 2
 92        self.layers = layers
 93        
 94        local_kwargs = locals()
 95        for name in _BLOCK_CONFIG:
 96            obj = local_kwargs[name]
 97            if not is_iterable(local_kwargs[name]):
 98                obj = [obj] * self.n_layers
 99            setattr(self, name, obj)
100        
101        self.validate()

104    def n_layers(self):
105        if not hasattr(self, 'layers'):
106            raise RuntimeError("n_layers called before layers initialized")
107        n_layers = len(self.layers)
108        # Since layers will be paired into n_in and n_out
109        # the number of layers will be equal to 1 for both
110        # length of 1 and 2, otherwise the number of layers
111        # will then be one less len(layers) because pairing
112        # starting from the first two elements
113        if n_layers > 1:
