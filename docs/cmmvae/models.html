171        """
172        Save the gradients of the model to TensorBoard.
173
174        This is a no-op if the number of named parameters exceeds gradient_record_cap to prevent clogging TensorBoard.
175        """

181                self.save_gradient(k, v.grad)

213        Get latent representations from AnnData.
214
215        Args:

287    def save_predictions(self, predictions: np.ndarray, idx: int):  
288        """
289        Accumulate predictions and save periodically dicated by :attr:`predict_save_interval`.
290        
291        Args:
292            predictions (np.ndarray): Predictions to be saved.
293            idx (int): Index of save prediction (To keep track of save interval).
294        """
295        self._running_predictions.append(predictions)
296        
297        div, mod = divmod(idx + 1, self.predict_save_interval)
298        if mod == 0:
299            self._save_paired_predictions()
300    
301    def _save_paired_predictions(self):
302        
303        self._curr_save_idx += 1
304        stacked_predictions = {}
305        
306        for key in self._running_predictions[0].keys():
307            if isinstance(self._running_predictions[0][key], pd.DataFrame):
308                stacked_predictions[key] = pd.concat([prediction[key] for prediction in self._running_predictions])
309            else:
310                stacked_predictions[key] = torch.cat([prediction[key] for prediction in self._running_predictions], dim=0).cpu().numpy()
311
312        for key in stacked_predictions:
313            if RK.METADATA in key:
314                continue
315            
316            self.save_latent_predictions(
317                embeddings=stacked_predictions[key], 
318                metadata=stacked_predictions[f"{key}_{RK.METADATA}"], 
319                embeddings_path=os.path.join(self.predict_dir, f"{key}_embeddings_{self._curr_save_idx}.npz"),
320                metadata_path=os.path.join(self.predict_dir, f"{key}_metadata_{self._curr_save_idx}.pkl"),
321            )

 86    def training_step(self, batch: Tuple[torch.Tensor, pd.DataFrame, str], batch_idx: int) -> None:
 87        """
 88        Perform a single training step.
 89
 90        This involves encoding the input, calculating losses, and updating weights.
 91
 92        Args:
 93            batch (tuple): Batch of data containing inputs, metadata, and expert ID.
 94            batch_idx (int): Index of the batch.
 95        """
 96        x, metadata, expert_id = batch
 97        expert_label = self.module.experts.labels[expert_id]
 98
 99        # Retrieve optimizers
100        optims = self.get_optimizers(zero_all=True)
101        expert_optimizer = optims['experts'][expert_id]
102        vae_optimizer = optims['vae']
103        adversarial_optimizers = optims['adversarials']
104
105        # Perform forward pass and compute the loss
106        qz, pz, z, xhats, cg_xhats, hidden_representations = self.module(x, metadata, expert_id)
107        
108        if x.layout == torch.sparse_csr:
109            x = x.to_dense()
110
111        # Calculate reconstruction loss
112        loss_dict = self.module.vae.elbo(qz, pz, x, xhats[expert_id], self.kl_annealing_fn.kl_weight)
113
114        # Train adversarial networks
115        adversarial_loss = None
116        for i, (hidden_rep, adv) in enumerate(zip(hidden_representations, self.module.adversarials)):
117            # Create expert labels
118            expert_labels = torch.full((self.batch_size,), expert_label, dtype=torch.long, device=x.device)
119
120            # Apply gradient reversal
121            reversed_hidden_rep = GradientReversalFunction.apply(hidden_rep, 0.1)
122            
123            # Get adversarial predictions
124            adv_output = adv(reversed_hidden_rep)
125
126            # Calculate adversarial loss
127            current_adversarial_loss = self.adversarial_criterion(adv_output, expert_labels)
128            if adversarial_loss is None:
129                adversarial_loss = current_adversarial_loss
130            else:
131                adversarial_loss += current_adversarial_loss
132
133            # Backpropagation for the adversarial
134            self.manual_backward(current_adversarial_loss, retain_graph=True)
135            adversarial_optimizers[i].step()
136            
137        loss_dict['adversarial_loss'] = adversarial_loss
138        loss = loss_dict[RK.LOSS]
139        
140        # Backpropagation for encoder and decoder
141        self.manual_backward(loss)
142
143        # Clip gradients for stability
144        self.clip_gradients(vae_optimizer, gradient_clip_val=0.5, gradient_clip_algorithm="norm")
145        self.clip_gradients(expert_optimizer, gradient_clip_val=0.5, gradient_clip_algorithm="norm")
146        
147        # Handle inactive conditional modules
148        inactive_modules = []
149        if self.module.vae.conditionals:
150            for conditional_layer in self.module.vae.conditionals.layers:
151                inactive = conditional_layer.unique_conditions - conditional_layer.active_condition_modules
152                inactive_modules.extend([conditional_layer.conditions[key] for key in conditional_layer.conditions if key in inactive])
153        
154        for module in inactive_modules:
155            for param in module.parameters():
156                if param.grad is not None:
157                    param.grad.zero_()
158        
159        # Update the weights
160        vae_optimizer.step()
161        expert_optimizer.step()
162        self.kl_annealing_fn.step()
163        
164        # Log the loss
165        self.auto_log(loss_dict, tags=[self.stage_name, expert_id])
166

246                optim.zero_grad()

248        def replace_indices_with_optimizers(mapping, optimizer_list):
249            if isinstance(mapping, dict):
250                return {key: replace_indices_with_optimizers(value, optimizer_list) for key, value in mapping.items()}
251            else:

275        }
276
277        optimizer_list = []
278        self.optimizer_map = convert_to_flat_list_and_map(optimizers, optimizer_list)
